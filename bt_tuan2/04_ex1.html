# Gii h phng trnh tuyn tnh
# 4x1 + 3x2 - 5x3 = 2
# -2x1 - 4x2 + 5x3 = 5
# 8x1 + 8x2 = -3

A = np.array([
    [4, 3, -5],
    [-2, -4, 5],
    [8, 8, 0]
])
y = np.array([2, 5, -3])

x = np.linalg.solve(A, y)
yy = np.dot(A, x)
diff = np.linalg.norm(yy - y)

print(f"+ Ma tran A: \n{A}")
print(f"+ Vector y: {y}")
print(f"+ Nghiem x: {x}")
print(f"+ Kiem tra: yy = Ax = {yy} ==> ||yy -y ||_2 = {diff: .2f}")

Bài 3. Tính gần đúng số pi dùng phương pháp Monte Carlo¶

V mt ng trn $C$ c bn knh $r$ ni tip mt hnh vung $C$, v t ngu nhin $N$ chm ln hnh vung. T l cc chm nm trong hnh trn trn tng s chm xp x bng din tch ca hnh trn chia cho hnh vung nh sau: $$ \frac{N_C}{N} \approx \frac{\mathbb{S}_C}{\mathbb{S}_N} $$ ni m $N_C$ l cc chm trong hnh trn, $N$ l tng s chm, $\mathbb{S}_C$ v $\mathbb{S}_N$ ln lt l din tch ca hnh trn v hnh vung. $$ \mathbb{S}_C = \pi * r^2 $$ $$ \mathbb{S}_N = (2*r)^2 = 4r^2 $$ Do , cng thc trn tr thnh: $$ \frac{N_C}{N} \approx \frac{\pi * r^2}{4r^2} = \frac{\pi}{4} $$ Suy ra, ta c gi tr s $\pi$ c tnh nh sau: $$ \pi \approx \frac{4N_C}{N} $$

Cc bn hy lp trnh tnh sp x s $\pi$ dng phng php Monte Carlo. Cho bit cc sai s vi $N = 100$, $N = 10000$, v $N = 1000000$. V th minh ha cho chng trnh.

Hng dn: Vit chng trnh pht sinh $N$ im $(x_i, y_i)$ vi $x_i, y_i \in [-1, -1]$. $(x_i, y_i)$ thuc hnh trn tm $(0, 0)$ bn knh 1 ni tip hnh vung c tm ti $(0, 0)$ di cnh 2 khi v ch khi $x_i^2 + y_i^2 <= 1$.

import numpy as np

def calc_pi_monte_carlo(n = 100):
    # Sinh ngu nhin n im (x, y) trong [-1, 1] x [-1, 1]
    x = np.random.uniform(-1, 1, n)
    y = np.random.uniform(-1, 1, n)
    # m s im nm trong hnh trn tm (0,0), bn knh 1
    inside = (x ** 2 + y ** 2) <= 1
    pi = 4 * np.sum(inside) / n
    return pi
